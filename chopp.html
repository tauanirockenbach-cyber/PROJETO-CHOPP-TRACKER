#include #include #include #include #include // 👈 NOVA BIBLIOTECA ADICIONADA AQUI // Chama as credenciais guardadas no ficheiro secrets.h #include "secrets.h" // ========================================================== // PINOS DO LEITOR RFID (RC522) NO ESP32 // ========================================================== #define RST_PIN 22 #define SS_PIN 5 MFRC522 mfrc522(SS_PIN, RST_PIN); // Variáveis de controle do sistema bool modoLeitura = false; unsigned long ultimoCheck = 0; void setup() { Serial.begin(115200); delay(1000); Serial.println("\n--- INICIANDO SISTEMA CERVEJARIA ---"); Serial.print("Conectando ao Wi-Fi: "); Serial.println(ssid); WiFi.begin(ssid, password); while (WiFi.status() != WL_CONNECTED) { delay(500); Serial.print("."); } Serial.println("\n✅ Wi-Fi Conectado!"); Serial.print("IP: "); Serial.println(WiFi.localIP()); SPI.begin(); mfrc522.PCD_Init(); delay(50); Serial.println("✅ Leitor RFID Pronto! Aguardando comandos do site..."); } void loop() { if (WiFi.status() != WL_CONNECTED) { Serial.println("⚠️ Wi-Fi desconectado! Tentando reconectar..."); WiFi.begin(ssid, password); delay(5000); return; } if (!modoLeitura) { if (millis() - ultimoCheck >= 1500) { ultimoCheck = millis(); String urlCheck = urlGoogle + "?acao=get_comando"; String comandoNuvem = enviarRequisicao(urlCheck); comandoNuvem.trim(); Serial.println("🔎 Status no Google: [" + comandoNuvem + "]"); if (comandoNuvem == "PENDENTE") { Serial.println("\n========================================="); Serial.println("📡 COMANDO RECEBIDO! APROXIME A TAG AGORA"); Serial.println("========================================="); modoLeitura = true; } } } else { if (mfrc522.PICC_IsNewCardPresent() && mfrc522.PICC_ReadCardSerial()) { String conteudo = ""; for (byte i = 0; i < mfrc522.uid.size; i++) { conteudo.concat(String(mfrc522.uid.uidByte[i] < 0x10 ? " 0" : " ")); conteudo.concat(String(mfrc522.uid.uidByte[i], HEX)); } conteudo.toUpperCase(); String uid = conteudo.substring(1); Serial.println("\n🍺 TAG LIDA COM SUCESSO: " + uid); mfrc522.PICC_HaltA(); mfrc522.PCD_StopCrypto1(); Serial.println("⏳ Enviando UID para a nuvem..."); String urlResult = urlGoogle + "?acao=resultado&resultado=SUCESSO&uid=" + uid; urlResult.replace(" ", "%20"); String respostaFinal = enviarRequisicao(urlResult); Serial.println("✅ Resposta do Servidor: " + respostaFinal); Serial.println("🔄 Retornando ao modo de espera...\n"); modoLeitura = false; ultimoCheck = millis(); } delay(50); } } // ========================================================== // FUNÇÃO PARA COMUNICAR COM O GOOGLE SCRIPT (AGORA COM SSL) // ========================================================== String enviarRequisicao(String url) { // 👈 MAGIA ACONTECE AQUI: Criamos um cliente que ignora bloqueios de certificado WiFiClientSecure client; client.setInsecure(); HTTPClient http; // Passamos o 'client' seguro junto com a URL http.begin(client, url); http.setFollowRedirects(HTTPC_STRICT_FOLLOW_REDIRECTS); int httpCode = http.GET(); String payload = ""; if (httpCode > 0) { payload = http.getString(); } else { Serial.println("❌ Erro HTTP: " + String(http.errorToString(httpCode).c_str())); payload = "ERRO"; } http.end(); return payload; }
